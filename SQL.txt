#获取去年12月时间
select to_char(add_months(trunc(sysdate,'year'),-1)'yyyymm') from dual;

#1.4.7.10月20日以后 oracle
SELECT *  FROM VSCMR_TCI_PAY_TYPE_MONTH
WHERE TO_CHAR(sysdate,'mm') IN ('01','04','07','10')
AND TO_CHAR(sysdate,'dd') >'20'
AND  date1 BETWEEN TO_CHAR(add_months(sysdate,-4),'yyyymm') 
AND  TO_CHAR(add_months(sysdate,-1),'yyyymm')
